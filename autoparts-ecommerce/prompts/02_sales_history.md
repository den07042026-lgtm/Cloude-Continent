ЧАСТЬ А — ЗАГРУЗКА ДАННЫХ:

  import pandas as pd
  from pathlib import Path

  артикул = "[артикул]"

  # Вариант 1: файл от оператора
  # df = pd.read_excel("[путь к файлу]")

  # Вариант 2: combined_marketplaces
  df = pd.read_parquet("data/processed/combined_marketplaces.parquet")
  df = df[df["offer_id"] == артикул].copy()

  if df.empty:
      print(f"Нет данных продаж для артикула {артикул}. Товар новый?")
  else:
      df["month_end"] = pd.to_datetime(df["month_end"])
      monthly = df.groupby(["month_end", "marketplace"]).agg(
          quantity=("quantity", "sum"),
          revenue=("fin_payout", "sum"),
          profit=("profit", "sum"),
      ).reset_index()
      print(monthly.to_string())
