  СЕБЕСТОИМОСТЬ = [из справочника]  # руб/шт

  # Вводные из конкурентов:
  МЕДИАНА_РЫНКА = [из Промпта 3]   # средняя реальная цена конкурентов
  МИНИ_КОНКУРЕНТ = [минимальная цена рынка]
  МАКСИ_КОНКУРЕНТ = [максимальная цена рынка]

  # Комиссии Ozon (для данной категории — уточнить в ЛК):
  КОМИССИЯ_ПРОЦЕНТ = 0.15   # 15% типичная для автозапчастей, скорректировать
  ЛОГИСТИКА_FBO = 80        # руб/шт (средняя для данного веса/габарита)
  ЛОГИСТИКА_FBS = 60        # руб/шт
  ЭКВАЙРИНГ = 0.015         # 1.5%

ЧАСТЬ Б — ЮНИТ-ЭКОНОМИКА ПРИ РАЗНЫХ ЦЕНАХ:

  цены = [цена for цена in range(
      int(МИНИ_КОНКУРЕНТ * 0.9),
      int(МАКСИ_КОНКУРЕНТ * 1.1),
      50  # шаг 50 руб
  )]

  rows = []
  for цена in цены:
      комиссия = цена * КОМИССИЯ_ПРОЦЕНТ
      эквайр = цена * ЭКВАЙРИНГ
      payout = цена - комиссия - ЛОГИСТИКА_FBO - эквайр
      прибыль = payout - СЕБЕСТОИМОСТЬ
      маржа = прибыль / payout * 100 if payout > 0 else 0
      позиция = "выше рынка" if цена > МЕДИАНА_РЫНКА * 1.05 else (
                "ниже рынка" if цена < МЕДИАНА_РЫНКА * 0.95 else "в рынке")
      rows.append({
          "Цена продажи": цена,
          "Комиссия Ozon": round(комиссия, 0),
          "Логистика FBO": ЛОГИСТИКА_FBO,
          "Эквайринг": round(эквайр, 0),
          "Payout": round(payout, 0),
          "Себестоимость": СЕБЕСТОИМОСТЬ,
          "Прибыль/шт": round(прибыль, 0),
          "Маржа %": round(маржа, 1),
          "Позиция": позиция,
      })
  df_ec = pd.DataFrame(rows)
  # Минимальная цена = где маржа ≥ 15%
  min_price = df_ec[df_ec["Маржа %"] >= 15]["Цена продажи"].min()
  print(f"Минимальная цена (маржа ≥15%): {min_price} ₽")
  print(df_ec.to_string())
