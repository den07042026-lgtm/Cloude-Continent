  }
  df.columns = [COLS.get(i, f"col_{i}") for i in range(len(df.columns))]
  df = df[df["Название"].notna() & (df["Название"].astype(str) != "nan")]

  # ВАЖНО: колонка col 20 "Объем товара, л" из Ozon Bestsellers содержит
  # некорректные значения (не объём одной единицы товара).
  # Объём ВСЕГДА извлекаем из названия товара через регулярное выражение.
  import re

  def extract_volume_liters(name):
      name = str(name)
      # Сначала ищем миллилитры: "750 мл", "750мл"
      m = re.search(r'(\d[\d\s]*[\.,]?\d*)\s*мл', name, re.IGNORECASE)
      if m:
          val = m.group(1).replace(' ', '').replace(',', '.')
          try:
              return round(float(val) / 1000, 4)
          except:
              pass
      # Затем литры: "1 л", "1л", "0,9 л", "1.5 л"
      m = re.search(r'(\d[\d]*[\.,]\d+|\d+)\s*л(?:\b|итр)', name, re.IGNORECASE)
      if m:
          val = m.group(1).replace(',', '.')
          try:
              v = float(val)
              if v < 20:  # защита от случайных чисел
                  return round(v, 4)
          except:
              pass
      return None

  df["nl"] = df["Название"].astype(str).str.lower()
  df["Объём_л"] = df["Название"].apply(extract_volume_liters)

ЧАСТЬ Б — ФИЛЬТРАЦИЯ (30 прямых конкурентов):

  МОЙ_ОБЪЁМ = [объём из карточки товара в литрах, например 1.0]

  # ПРИНЦИП ФИЛЬТРАЦИИ: конкуренты должны совпадать по форме выпуска,
  # объёму И ключевым характеристикам товара (из карточки Промпта 1).
  # Например: если товар — жидкий ультраконцентрат парфюмированный 1 л,
  # то гранулы, капсулы, саше, листы — НЕ конкуренты, даже если попадают
  # в ту же категорию Ozon.

  # 1. Исключить нежидкие / нерелевантные форматы
  #    → подставить ключевые слова исходя из формата своего товара
  EXCLUDE_FORMAT = [
      # для жидкого кондиционера для белья:
      "гранул", "капсул", "саше", "листовой", "лист ", "таблетк",
      "полоск", "стик", "порошок", "шарик",
      # для других категорий — заменить на своё
  ]
  mask_bad_format = df["nl"].str.contains("|".join(EXCLUDE_FORMAT), na=False)

  # 2. По объёму: диапазон ±50–100% от моего объёма
  #    (для концентратов малого объёма — жёсткий диапазон, например 0.75–2.0 л при МОЙ ОБЪЁМ=1.0)
  mask_volume = df["Объём_л"].between(МОЙ_ОБЪЁМ * 0.75, МОЙ_ОБЪЁМ * 2.0)

  # 3. Ключевые характеристики товара (из Промпта 1)
  #    → включить совпадающие свойства (парфюм, концентрат, назначение и т.п.)
  INCLUDE_PROPS = [
      # для парфюмированного концентрата:
      "парфюм", "parfum", "аромат", "fragrance",
      "концентрат", "concentrate", "ультраконцентрат",
      # для других товаров — заменить на своё
  ]
  mask_props = df["nl"].str.contains("|".join(INCLUDE_PROPS), na=False)

  # 4. Применить фильтры
  filtered = df[~mask_bad_format & mask_volume & mask_props].copy()

  # 4. Сортировка по выручке, топ-30
  filtered = filtered.sort_values("Заказано_руб", ascending=False).head(30).reset_index(drop=True)
  filtered["№"] = filtered.index + 1

  # 5. Цена за объём как у нашего товара (для честного сравнения)
  col_ekv = f"Цена_за_{МОЙ_ОБЪЁМ}л"
  filtered[col_ekv] = (
      filtered["Средняя_цена"] * МОЙ_ОБЪЁМ / filtered["Объём_л"]
  ).round(0)
  # Для строк без объёма — ставим Средняя_цена как есть
  filtered.loc[filtered["Объём_л"].isna(), col_ekv] = filtered["Средняя_цена"]
