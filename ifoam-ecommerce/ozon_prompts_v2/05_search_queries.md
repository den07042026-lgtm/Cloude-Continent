  import sys
  sys.path.insert(0, "C:/Users/1/Downloads/ozon_prompts_v2")
  from pathlib import Path

  fetcher = Path("C:/Users/1/Downloads/ozon_prompts_v2/05_seo_fetcher.py")
  content = fetcher.read_text(encoding="utf-8")

  # Заменить блок SEARCH_QUERIES
  import re
  new_queries = """SEARCH_QUERIES = [
      [СПИСОК_ЗАПРОСОВ_В_КАВЫЧКАХ],
  ]"""
  new_group = 'GROUP_NAME = "[ГРУППА]"'

  content = re.sub(r'SEARCH_QUERIES = \[.*?\]', new_queries, content, flags=re.DOTALL)
  content = re.sub(r'GROUP_NAME = ".*?"', new_group, content)
  fetcher.write_text(content, encoding="utf-8")
  print("Обновлено. Запусти скрипт.")
  print(f"Запросы: [СПИСОК]")
  print(f"Группа: [ГРУППА]")
